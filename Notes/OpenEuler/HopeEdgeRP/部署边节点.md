```shell
# 1. 克隆源代码
git clone
# 2. 进入项目根目录，配置 Python 虚拟环境
python3 -m venv venv
# 3. 激活虚拟环境
source venv/bin/activate
# 4. 安装项目依赖库
python3 -m pip install -r requirement.txt  --trusted-host=pypi.python.org --trusted-host=pypi.org --trusted-host=files.pythonhosted.org --trusted-host=pypi.tuna.tsinghua.edu.cn -i https://pypi.tuna.tsinghua.edu.cn/simple/
```
